17#[derive(Clone)]
18struct TimerCallback {
19    callback: *mut c_void,
20    argument: sigval,
21}
22
23struct SigTimer {
24    callback: TimerCallback,
25    _clk: ctypes::clockid_t,
26    exited: bool,
27    /// The next deadline of the timer.
28    deadline: u64,
29    /// The interval of the timer(nanoseconds).
30    ///
31    /// If the interval is 0, the timer is a one-shot timer.
32    interval: u64,
33    wq: Arc<WaitQueue>,
34    task: AxTaskRef,
35}
36
37unsafe impl Send for SigTimer {}
38unsafe impl Sync for SigTimer {}
39
40static TIMER_MAP: Mutex<BTreeMap<usize, SigTimer>> = Mutex::new(BTreeMap::new());
41
42fn create_timer_thread(callback_fn: *mut c_void, argument: sigval) -> AxTaskRef {
43    let callback_fn = unsafe { core::mem::transmute::<*mut c_void, fn(sigval)>(callback_fn) };
44    let arg_addr = &argument as *const _ as usize;
45    axtask::spawn(move || {
46        let arg = unsafe { core::ptr::read_unaligned(arg_addr as *const sigval) };
47        callback_fn(arg);
48        axtask::exit(0);
49    })
50}

52/// Create a timer.
53pub unsafe fn sys_timer_create(
54    clk: ctypes::clockid_t,
55    se: *const ctypes::sigevent,
56    timer: *mut ctypes::timer_t,
57) -> c_int {
58    static TIMER_ID: AtomicUsize = AtomicUsize::new(1);
59    syscall_body!(sys_timer_create, {
60        if clk != ctypes::CLOCK_REALTIME as c_int {
61            return Err(LinuxError::ENOSYS);
62        }
63        const SIGEV_THREAD: c_int = 2;
64        let se = unsafe { &*se };
65        if se.sigev_notify != SIGEV_THREAD {
66            return Err(LinuxError::ENOSYS);
67        }
68
69        let id = TIMER_ID.fetch_add(1, Ordering::Relaxed);
70        let wq = Arc::new(WaitQueue::new());
71        let wq_ptr = Arc::clone(&wq);
72        let task = axtask::spawn(move || {
73            fn deadline(timer_id: usize) -> u64 {
74                TIMER_MAP.lock().get(&timer_id).unwrap().deadline
75            }
76            if TIMER_MAP.lock().get(&id).is_none() {
77                axtask::WaitQueue::wait(&wq_ptr);
78            }
79            loop {
80                if TIMER_MAP.lock().get(&id).unwrap().exited {
81                    break;
82                }
83                loop {
84                    let curr_ddl = deadline(id);
85                    let dur = curr_ddl - axhal::time::wall_time_nanos();
86                    if curr_ddl == 0 || dur as i64 <= 0 {
87                        break;
88                    }
89                    axtask::WaitQueue::wait_timeout(&wq_ptr, TimeValue::from_nanos(dur));
90                }
91
92                let curr_ddl = deadline(id);
93                if curr_ddl != 0 && axhal::time::wall_time_nanos() >= curr_ddl {
94                    // The timer has been expired, so handle the callback.
95                    let handler = TIMER_MAP.lock().get_mut(&id).unwrap().callback.clone();
96                    let task = create_timer_thread(handler.callback, handler.argument);
97                    let tid = task.id().as_u64();
98                    let pthread = Pthread::from_axtask(task);
99
100                    TID_TO_PTHREAD.write().insert(tid, Arc::new(pthread));
101                    Pthread::join(tid as _).unwrap();
102
103                    // TODO: apply the interval to the next deadline.
104                }
105            }
106
107            axtask::exit(0);
108        });
109
110        TIMER_MAP.lock().insert(
111            id,
112            SigTimer {
113                callback: TimerCallback {
114                    callback: unsafe {
115                        se.__sev_fields.__sev_thread.sigev_notify_function.unwrap() as *mut c_void
116                    },
117                    argument: se.sigev_value,
118                },
119                _clk: clk,
120                exited: false,
121                deadline: 0,
122                interval: 0,
123                wq: Arc::clone(&wq),
124                task,
125            },
126        );
127
128        unsafe {
129            let timer = &mut *timer;
130            *timer = id as ctypes::timer_t;
131        }
132        Ok(0)
133    })
134}

136/// Set the time for a timer.
137pub unsafe fn sys_timer_settime(
138    timer: ctypes::timer_t,
139    flags: c_int,
140    new_value: *const itimerspec,
141    old_value: *mut itimerspec,
142) -> c_int {
143    fn to_nanos(ts: timespec) -> u64 {
144        ts.tv_nsec as u64 + ts.tv_sec as u64 * axhal::time::NANOS_PER_SEC
145    }
146
147    syscall_body!(sys_timer_settime, {
148        if flags != 0 || !old_value.is_null() {
149            warn!("Not implemented: timer_settime with flags != 0 or old_value != NULL");
150            return Err(LinuxError::EINVAL);
151        }
152        if new_value.is_null() {
153            return Err(LinuxError::EINVAL);
154        }
155        let mut timer_map = TIMER_MAP.lock();
156        let timer = timer_map
157            .get_mut(&(timer as usize))
158            .ok_or(LinuxError::EINVAL)?;
159        let new_value = unsafe { &*new_value };
160        if new_value.it_interval.tv_sec != 0 || new_value.it_interval.tv_nsec != 0 {
161            warn!("Not implemented: timer_settime with it_interval != 0");
162            return Err(LinuxError::EINVAL);
163        }
164        // The value is relative to the current time.
165        timer.deadline = axhal::time::wall_time_nanos() + to_nanos(new_value.it_value);
166        timer.interval = to_nanos(new_value.it_interval);
167        timer.wq.notify_one(false);
168        Ok(0)
169    })
170}

172/// Delete a timer.
173pub unsafe fn sys_timer_delete(timer: ctypes::timer_t) -> c_int {
174    syscall_body!(sys_timer_delete, {
175        let task_to_join = {
176            let mut timer_map = TIMER_MAP.lock();
177            timer_map.get_mut(&(timer as usize)).map(|timer_data| {
178                timer_data.deadline = 0;
179                timer_data.exited = true;
180                timer_data.wq.notify_one(false);
181                timer_data.task.clone()
182            })
183        };
184
185        if let Some(task) = task_to_join {
186            task.join().unwrap();
187        }
188
189        TIMER_MAP.lock().remove(&(timer as usize));
190
191        Ok(0)
192    })
193}
