1//! Provides the corresponding malloc(size_t) and free(size_t) when using the C user program.
2//!
3//! The normal malloc(size_t) and free(size_t) are provided by the library malloc.h, and
4//! sys_brk is used internally to apply for memory from the kernel. But in a unikernel like
5//! `ArceOS`, we noticed that the heap of the Rust user program is shared with the kernel. In
6//! order to maintain consistency, C user programs also choose to share the kernel heap,
7//! skipping the sys_brk step.
8
9use crate::ctypes;
10use arceos_posix_api as api;
11
12use core::ffi::c_void;
13use core::ptr::NonNull;
14
15/// Allocate memory and return the memory address.
16///
17/// Returns 0 on failure (the current implementation does not trigger an exception)
18#[unsafe(no_mangle)]
19pub unsafe extern "C" fn malloc(size: ctypes::size_t) -> *mut c_void {
20    api::sys_malloc(size).as_ptr() as *mut c_void
21}
22
23/// Deallocate memory.
24///
25/// (WARNING) If the address to be released does not match the allocated address, an error should
26/// occur, but it will NOT be checked out. This is due to the global allocator `Buddy_system`
27/// (currently used) does not check the validity of address to be released.
28#[unsafe(no_mangle)]
29pub unsafe extern "C" fn free(ptr: *mut c_void) {
30    if ptr.is_null() {
31        return;
32    }
33    api::sys_free(NonNull::new(ptr as *mut u8).unwrap());
34}
35
36/// Allocate pages
37#[unsafe(no_mangle)]
38pub unsafe extern "C" fn page_alloc(size: ctypes::size_t) -> *mut c_void {
39    api::sys_page_alloc(size).as_ptr() as *mut c_void
40}
41
42/// Free pages
43#[unsafe(no_mangle)]
44pub unsafe extern "C" fn page_free(ptr: *mut c_void, size: ctypes::size_t) {
45    api::sys_page_free(NonNull::new(ptr as *mut u8).unwrap(), size);
46}
